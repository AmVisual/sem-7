# Zamiana danych na macierz
tr_matrix = data_tr.to_numpy()
# Normalizacja wartoci redniej ROI0 (moc sygnau) - obszar poza obrazowanego obiektu
# Moc sygnau spada dla wyszych wartoci czasu TR, obserwujemy to dla redniej wartoci ROI0.
tr_matrix[:,1] = tr_matrix[:,1] / max(tr_matrix[:,1])
# Przeliczenie wartoci rednich (mean) i odchyle standardowy (sd) w zalenoci od mocy sygnau wzgldem pomiaru dla TR=50 ms
for i in range(3,13):
    tr_matrix[:,i] = tr_matrix[:,i] / tr_matrix[:,1]

# Wykres zalenoci amplitudy sygnau od czasu
tr = tr_matrix[:,0]

    plt.errorbar(tr, tr_matrix[:,i], yerr=tr_matrix[:,i+1], label="ROI"+str(int((i-1)/2)),capsize=2.5)

plt.show()
